CREATE TABLE medecin(
    idM INT PRIMARY KEY AUTO_INCREMENT,
    nomM VARCHAR(30) NOT NULL,
    prenomM VARCHAR(30) NOT NULL,
    dnM DATE NOT NULL,
    specialite VARCHAR(50) NOT NULL
);

CREATE TABLE patient(
    idP INT PRIMARY KEY AUTO_INCREMENT,
    nomP VARCHAR(30) NOT NULL,
    prenomP VARCHAR(30) NOT NULL,
    dnP DATE NOT NULL,
    tel CHAR(12) NOT NULL,
    email VARCHAR(30) NOT NULL,
    maladie VARCHAR(50) NOT NULL
);

CREATE TABLE rv (
    idRV INT PRIMARY KEY AUTO_INCREMENT,
    idM INT REFERENCES medecin(idM),
    idP INT REFERENCES patient(idP),
    dateRV DATETIME NOT NULL
);

CREATE TABLE contact (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nom VARCHAR(30) NOT NULL,
    prenom VARCHAR(30) NOT NULL,
    Email VARCHAR(50) NOT NULL,
    message VARCHAR(100) NOT NULL
);
